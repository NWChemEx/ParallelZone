50template<typename T>
51BinaryBuffer make_binary_buffer(T&& input) {
52 using clean_type = std::decay_t<T>;
53 if constexpr(needs_serialized_v<clean_type>) {
54 using pimpl_type = detail_::BinaryBufferPIMPL<std::string>;
55 // TODO: Put directly into a string
56 std::stringstream ss;
57 {
58 cereal::BinaryOutputArchive ar(ss);
59 ar << std::forward<T>(input);
60 }
61 auto pimpl = std::make_unique<pimpl_type>(ss.str());
62 return BinaryBuffer(std::move(pimpl));
63 } else {
64 using pimpl_type = detail_::BinaryBufferPIMPL<clean_type>;
65 auto pimpl = std::make_unique<pimpl_type>(std::forward<T>(input));
66 return BinaryBuffer(std::move(pimpl));
67 }
68}

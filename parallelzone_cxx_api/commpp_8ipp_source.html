68template<typename T>
69typename CommPP::gather_return_type<T> CommPP::gather_t_(
70 T&& input, opt_root_t root) const {
71 using clean_type = std::decay_t<T>;
72 using return_type = typename CommPP::gather_return_type<clean_type>;
73 using value_type = typename return_type::value_type;
74
75 const bool am_i_root = root.has_value() ? me() == *root : true;
76
77 if constexpr(needs_serialized_v<clean_type>) {
78 // Do gather in binary
79 auto binary = make_binary_buffer(std::forward<T>(input));
80 auto binary_rv = gather_(binary, root);
81
82 // Early out if not root
83 return_type rv;
84 if(!binary_rv.has_value()) return rv;
85
86 // We got back a std::vector<std::byte> which contains this->size()
87 // instances of type clean_type, the serialized form of each clean_type
88 // object has size binary_rv->size() / this->size()
89 const auto& buffer = *binary_rv;
90 auto serialized_size = buffer.size() / size();
91
92 value_type vec(size());
93 for(size_type i = 0, j = 0; i < size_type(buffer.size());
94 i += serialized_size, ++j) {
95 const_binary_reference view(binary_rv->data() + i, serialized_size);
96 vec[j] = std::move(from_binary_view<clean_type>(view));
97 }
98 rv.emplace(std::move(vec));
99 return rv;
100 } else {
101 // TODO: make traits to_binary, binary_size to wrap calling .data() and
102 // .size()
103 const_binary_reference input_binary(input.data(), input.size());
104
105 // Make output buffer, but only allocate on root
106 value_type output;
107 if(am_i_root) value_type(input.size() * size()).swap(output);
108 binary_reference output_binary(output.data(), output.size());
109
110 gather_(input_binary, output_binary, root);
111
112 return_type rv;
113 if(am_i_root) rv.emplace(std::move(output));
114 return rv;
115 }
116}

118template<typename T>
119typename CommPP::gather_return_type<T> CommPP::gatherv_t_(
120 T&& input, opt_root_t root) const {
121 using clean_type = std::decay_t<T>;
122 using return_type = typename CommPP::gather_return_type<clean_type>;
123 using value_type = typename return_type::value_type;
124
125 if constexpr(needs_serialized_v<clean_type>) {
126 // Do gather in binary
127 auto binary = make_binary_buffer(std::forward<T>(input));
128 auto binary_rv = gatherv_(binary, root);
129
130 // Early out if not root
131 return_type rv;
132 if(!binary_rv.has_value()) return rv;
133
134 // We got back a std::vector<std::byte> of the binary data
135 // and the sizes sent by each rank
136 const auto& buffer = binary_rv->first;
137 const auto& sizes = binary_rv->second;
138
139 value_type vec(size());
140 for(size_type i = 0, total = 0; i < size(); ++i) {
141 const_binary_reference view(buffer.data() + total, sizes[i]);
142 vec[i] = std::move(from_binary_view<clean_type>(view));
143 total += sizes[i];
144 }
145 rv.emplace(std::move(vec));
146 return rv;
147 } else {
148 // TODO: make traits to_binary, binary_size to wrap calling .data() and
149 // .size()
150 const_binary_reference input_binary(input.data(), input.size());
151
152 // TODO: avoid copy by preallocating buffer
153 auto binary_rv = gatherv_(input_binary, root);
154
155 // Early out if not root
156 return_type rv;
157 if(!binary_rv.has_value()) return rv;
158
159 // We got back a std::vector<std::byte> of the binary data
160 // and the sizes (in bytes) sent by each rank
161 const auto& buffer = binary_rv->first;
162 using element_type = typename value_type::value_type;
163 constexpr auto t_size = sizeof(element_type);
164
165 value_type vec;
166 for(size_type i = 0; i < size_type(buffer.size()); i += t_size) {
167 const_binary_reference view(buffer.data() + i, t_size);
168 vec.emplace_back(std::move(from_binary_view<element_type>(view)));
169 }
170 rv.emplace(std::move(vec));
171 return rv;
172 }

175template<typename T, typename Fxn>
176typename CommPP::reduce_return_type<T> CommPP::reduce_t_(
177 T&& input, Fxn&& fxn, opt_root_t root) const {
178 // Assumed to be a container
179 using clean_type = std::decay_t<T>;
180 using value_type = typename clean_type::value_type;
181 using clean_fxn = std::decay_t<Fxn>;
182
183 static_assert(!needs_serialized_v<clean_type>, "Doesn't needs serialized?");
184 static_assert(has_mpi_data_type_v<value_type>, "Is a recognized MPI type?");
185 static_assert(has_mpi_op_v<clean_fxn>, "Is a recognized MPI Operation?");
186
187 const auto am_i_root = root.has_value() ? me() == *root : true;
188 const auto n_elems = input.size();
189 auto type = mpi_data_type_v<value_type>;
190 auto op = mpi_op_v<clean_fxn>;
191
192 std::vector<value_type> temp;
193 if(am_i_root) std::vector<value_type>(n_elems).swap(temp);
194
195 const auto send = input.data();
196 const auto recv = temp.data();
197
198 if(root.has_value()) {
199 MPI_Reduce(send, recv, n_elems, type, op, *root, comm());
200 } else {
201 MPI_Allreduce(send, recv, n_elems, type, op, comm());
202 }
203 reduce_return_type<T> rv;
204 if(am_i_root) rv.emplace(std::move(temp));
205 return rv;
206}

71void RuntimeViewPIMPL::instantiate_resource_set_(size_type rank) const {
72 using rs_pimpl = detail_::ResourceSetPIMPL;
73 if(m_resource_sets_.count(rank)) return;
74
75 // Null loggers for now
76 logger_type logger;
77
78 auto p = std::make_unique<rs_pimpl>(rank, m_comm, std::move(logger));
79 m_resource_sets_.emplace(rank, ResourceSet(std::move(p)));
80}

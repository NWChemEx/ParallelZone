90template<typename FxnType, typename... Args>
91auto make_inner_lambda_(FxnType&& fxn, Args&&... args) {
92 auto wrapped_args = wrap_args_(std::forward<Args>(args)...);
93 auto l = [fxn = std::forward<FxnType>(fxn),
94 args = std::move(wrapped_args)]() mutable {
95 return apply_(fxn, std::move(args));
96 };
97 return l;

119template<typename FxnType, typename... Args>
120static auto make_outer_lambda_(FxnType&& fxn, Args&&... args) {
121 auto inner_lambda = make_inner_lambda_(std::forward<FxnType>(fxn),
122 std::forward<Args>(args)...);
123
124 using result_type = decltype(inner_lambda());
125 if constexpr(std::is_same_v<result_type, void>) {
126 return [inner_lambda = std::move(inner_lambda)]() mutable {
127 inner_lambda();
128 return std::any{};
129 };
130 } else {
131 return [inner_lambda = std::move(inner_lambda)]() mutable {
132 return std::make_any<result_type>(inner_lambda());
133 };
134 }
